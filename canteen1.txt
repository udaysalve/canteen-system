    <script>
        document.addEventListener('DOMContentLoaded', () => {

            const addButtons = document.querySelectorAll('.add-btn');
            const orderDetails = document.getElementById('order-details');
            const viewOrderBtn = document.getElementById('view-order-btn');

            let cart = [];

            addButtons.forEach(button => {
                button.addEventListener('click', (event) => {
                    const menuItem = event.target.closest('.menu-item');
                    const itemName = menuItem.dataset.name;
                    const itemPrice = parseFloat(menuItem.dataset.price);

                    addToCart(itemName, itemPrice);
                });
            });

            function addToCart(name, price) {
                // Check if item already exists in cart to increase quantity, or add it as a new item
                const existingItem = cart.find(item => item.name === name);
                if (existingItem) {
                    existingItem.quantity++;
                } else {
                    cart.push({ name, price, quantity: 1 });
                }
                
                updateOrderSummary();
            }

            function updateOrderSummary() {
                if (cart.length === 0) {
                    orderDetails.innerHTML = '<p>Your order is empty</p>';
                    viewOrderBtn.classList.add('hidden');
                    return;
                }

                let totalItems = 0;
                let totalPrice = 0;

                cart.forEach(item => {
                    totalItems += item.quantity;
                    totalPrice += item.price * item.quantity;
                });

                orderDetails.innerHTML = `
                    <p>${totalItems} Item${totalItems > 1 ? 's' : ''} | ₹ ${totalPrice.toFixed(2)}</p>
                `;
                viewOrderBtn.classList.remove('hidden');
            }

        });
    </script>
